CREATE OR REPLACE TRIGGER MAIN.R1GE0008
BEFORE INSERT OR UPDATE ON MAIN.TAGE0008
FOR EACH ROW
BEGIN
  IF INSERTING OR (UPDATING AND :NEW.TAGE0008_ID IS NULL) THEN
      IF :NEW.GRUPTIPI = 4 THEN
         :NEW.TAGE0008_ID := :NEW.ANAGRUP||'-'||:NEW.ALTGRUP1||'-'||:NEW.ALTGRUP2||'-'||:NEW.ALTGRUP3;
      ELSIF :NEW.GRUPTIPI = 3 THEN
         :NEW.TAGE0008_ID := :NEW.ANAGRUP||'-'||:NEW.ALTGRUP1||'-'||:NEW.ALTGRUP2;
      ELSIF :NEW.GRUPTIPI = 2 THEN
         :NEW.TAGE0008_ID := :NEW.ANAGRUP||'-'||:NEW.ALTGRUP1;
      ELSIF :NEW.GRUPTIPI = 1 THEN
         :NEW.TAGE0008_ID := :NEW.ANAGRUP;
      END IF;
  END IF;
END;
